Quantify Scheduler
def qubit_spectroscopy_schedule():
    schedule = Schedule('qubit_spectroscopy', repetitions=1024)

    for qubit_frequency in frequencies:
        schedule.add(Reset('q12'))
        
        schedule.add(
            SoftSquarePulse(
                duration=2e-6,
                amp=6e-4,
                port='q12:mw',
                clock='q12.01'
            )
        )

        ro = schedule.add(
            SquarePulse(
                duration=3e-6,
                amp=3e-3,
                port='q12:res',
                clock='q12.ro'
            )
        )

        schedule.add(
            SSBIntegrationComplex(
                duration=2.8e-6,
                port='q12:res',
                clock='q12.ro',
            ),
            ref_op=ro,
            ref_pt="start",
            rel_time=200e-9
        )
    return schedule
